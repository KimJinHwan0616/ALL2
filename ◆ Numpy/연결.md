```
# (3, 3)
a = [ [1, 2, 3],
      [3, 4, 5],
      [7, 8, 9] ]

# (3, 2)
b = [ [5, 6], 
      [7, 8], 
      [1, 2] ]

# (2, 3)
c = [ [3, 2, 1],
      [9, 8, 7] ]
```

## 축 고정
차원 동일X
```angular2html
np.concatenate(
    (리스트1, ..., 리스트n),
    axis = 축,
    dtype = None
    )
```
+ 
    ```
    np.concatenate((a, c), axis=0)
    
    # (3+2, 3)
    array([[1, 2, 3],
           [3, 4, 5],
           [7, 8, 9],
           [3, 2, 1],
           [9, 8, 7]])
    ```
+
    ```
    np.concatenate((a, b), axis=1)
    
    # (3, 2+3)
    array([[1, 2, 3, 5, 6],
           [3, 4, 5, 7, 8],
           [7, 8, 9, 1, 2]])
    ```


## 축 생성
차원 동일O
```angular2html
np.stack(
    (리스트1, ..., 리스트n),
    axis = 축,
    dtype = None
    )
```
+ 
    ```
    np.stack((a, a), axis = 0)
    
    # (2, 3, 3)
    array([[[1, 2, 3],
            [3, 4, 5],
            [7, 8, 9]],
    
           [[1, 2, 3],
            [3, 4, 5],
            [7, 8, 9]]])
    ```
+ 
    ```
    np.stack((a, a), axis = 1)
    
    # (3, 2, 3)
    array([[[1, 2, 3],
            [1, 2, 3]],
    
           [[3, 4, 5],
            [3, 4, 5]],
    
           [[7, 8, 9],
            [7, 8, 9]]])
    ```
+ 
    ```
    np.stack((a, a), axis = 2)
    
    # (3, 3, 2)
    array([[[1, 1],
            [2, 2],
            [3, 3]],
    
           [[3, 3],
            [4, 4],
            [5, 5]],
    
           [[7, 7],
            [8, 8],
            [9, 9]]])
    ```
