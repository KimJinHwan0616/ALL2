## 구조체 
```angular2html
// 정의 // 
struct 구조체
{
    멤버변수;
};

/////////////////////////////////////

// 선언 + 초기화(Case 1) //
구조체 구조체_이름 = {값1, ..., 값n};

// 선언 + 초기화(Case 2) //
구조체 구조체_이름;
구조체_이름.멤버변수_이름 = 값;

// 호출 //
구조체_이름.멤버변수_이름;
```
```
// 선언 //
struct Fruit {
    string name;
    string color;
    int price;
};

/////////////////////////////////////

// 선언 + 초기화(Case 1) //
Fruit apple = {"Apple", "Red", 1000};

// 선언 + 초기화(Case 2) ★ // 
Fruit orange;
orange.name = "Orange";
orange.color = "Orange";
orange.price = 800;

// 호출 //
apple.name;    // Apple
apple.color;    // Red

orange.name;    // Orange
orange.price;    // 800
```
