## 기본 클래스
```angular2html
class 기본_클래스
{         
public:
    생성자
    virtual ~기본_클래스();    // 소멸자

    virtual 멤버함수;    // 가상함수
    virtual 멤버함수 = 0;    // 순수 가상함수
};

private:
    ...
};
```

## 파생 클래스
```angular2html
class 파생_클래스 : public 기본_클래스
{         
public:
    생성자
    ~파생_클래스();    // 소멸자

    멤버함수 override;    // 오버라이드 확인
};

private:
    ...
};
```
